```csharp
namespace StellarCommand.Core
{
    using System;
    using Firebase;
    using Firebase.Auth;
    using UnityEngine;

    public class FirebaseAuthManager : MonoSingleton<FirebaseAuthManager>
    {
        private FirebaseAuth _auth;
        private FirebaseUser _user;

        public string UserId => _user?.UserId;
        public bool IsAuthenticated => _user != null;
        public bool IsAnonymous => _user?.IsAnonymous ?? true;

        public event Action<bool> OnAuthStateChanged; // true = authenticated

        /// <summary>
        /// Initialize Firebase and sign in. Call from BootManager.Start().
        /// If Firebase is unavailable, logs warning and returns (game works offline).
        /// </summary>
        public async Awaitable InitializeAsync()
        {
            try
            {
                var status = await FirebaseApp.CheckAndFixDependenciesAsync();
                if (status != DependencyStatus.Available)
                {
                    Debug.LogWarning($"[FirebaseAuth] Dependencies not available: {status}. Running offline.");
                    return;
                }

                _auth = FirebaseAuth.DefaultInstance;
                _auth.StateChanged += HandleAuthStateChanged;

                if (_auth.CurrentUser != null)
                {
                    _user = _auth.CurrentUser;
                    Debug.Log($"[FirebaseAuth] Existing session: {_user.UserId} (anonymous={_user.IsAnonymous})");
                }
                else
                {
                    var result = await _auth.SignInAnonymouslyAsync();
                    _user = result.User;
                    Debug.Log($"[FirebaseAuth] Anonymous sign-in: {_user.UserId}");
                }

                OnAuthStateChanged?.Invoke(true);
            }
            catch (Exception e)
            {
                Debug.LogWarning($"[FirebaseAuth] Init failed: {e.Message}. Running offline.");
            }
        }

        /// <summary>
        /// Link anonymous account to Google credential.
        /// UID is preserved by Firebase — same database path, no data migration.
        /// Per D-02 and FOUND-13.
        /// </summary>
        public async Awaitable<bool> LinkWithGoogleAsync(string idToken)
        {
            if (_user == null)
            {
                Debug.LogError("[FirebaseAuth] No user to link. Auth not initialized.");
                return false;
            }

            try
            {
                var credential = GoogleAuthProvider.GetCredential(idToken, null);
                var result = await _user.LinkWithCredentialAsync(credential);
                _user = result.User;
                Debug.Log($"[FirebaseAuth] Linked to Google: {_user.UserId} (anonymous={_user.IsAnonymous})");
                OnAuthStateChanged?.Invoke(true);
                return true;
            }
            catch (Exception e)
            {
                Debug.LogError($"[FirebaseAuth] Link failed: {e.Message}");
                return false;
            }
        }

        /// <summary>
        /// Sign in with existing Google credential (for restore on reinstall — FOUND-14).
        /// </summary>
        public async Awaitable<bool> SignInWithGoogleAsync(string idToken)
        {
            try
            {
                var credential = GoogleAuthProvider.GetCredential(idToken, null);
                var result = await _auth.SignInWithCredentialAsync(credential);
                _user = result.User;
                Debug.Log($"[FirebaseAuth] Google sign-in: {_user.UserId}");
                OnAuthStateChanged?.Invoke(true);
                return true;
            }
            catch (Exception e)
            {
                Debug.LogError($"[FirebaseAuth] Google sign-in failed: {e.Message}");
                return false;
            }
        }

        private void HandleAuthStateChanged(object sender, EventArgs e)
        {
            if (_auth != null)
            {
                _user = _auth.CurrentUser;
            }
        }

        protected override void OnDestroy()
        {
            if (_auth != null)
            {
                _auth.StateChanged -= HandleAuthStateChanged;
            }
            base.OnDestroy();
        }
    }
}
```

```csharp
namespace StellarCommand.Core
{
    using System;
    using Firebase.Database;
    using UnityEngine;

    public class FirebaseSyncManager : MonoSingleton<FirebaseSyncManager>
    {
        private DatabaseReference _dbRef;
        private bool _isInitialized;

        /// <summary>
        /// Initialize with authenticated user ID. Call after FirebaseAuthManager.InitializeAsync().
        /// Database path: users/{userId}/save
        /// Per D-01: Firebase Realtime Database, single JSON blob per user.
        /// </summary>
        public void Initialize(string userId)
        {
            if (string.IsNullOrEmpty(userId))
            {
                Debug.LogWarning("[FirebaseSync] No userId — sync disabled.");
                return;
            }

            try
            {
                _dbRef = FirebaseDatabase.DefaultInstance
                    .GetReference("users")
                    .Child(userId)
                    .Child("save");
                _isInitialized = true;
                Debug.Log($"[FirebaseSync] Initialized for user: {userId}");
            }
            catch (Exception e)
            {
                Debug.LogWarning($"[FirebaseSync] Init failed: {e.Message}");
                _isInitialized = false;
            }
        }

        /// <summary>
        /// Push local save data to Firebase. Best-effort — failures are logged, not thrown.
        /// </summary>
        public async Awaitable PushSaveAsync(GameSaveData data)
        {
            if (!_isInitialized || _dbRef == null) return;

            try
            {
                string json = JsonUtility.ToJson(data);
                await _dbRef.SetRawJsonValueAsync(json);
                Debug.Log("[FirebaseSync] Save pushed to cloud.");
            }
            catch (Exception e)
            {
                Debug.LogWarning($"[FirebaseSync] Push failed (will retry next save): {e.Message}");
            }
        }

        /// <summary>
        /// Pull save data from Firebase. Returns null on failure.
        /// Used for restore-on-reinstall (FOUND-14).
        /// </summary>
        public async Awaitable<GameSaveData> PullSaveAsync()
        {
            if (!_isInitialized || _dbRef == null) return null;

            try
            {
                var snapshot = await _dbRef.GetValueAsync();
                if (!snapshot.Exists)
                {
                    Debug.Log("[FirebaseSync] No cloud save found.");
                    return null;
                }

                string json = snapshot.GetRawJsonValue();
                var data = JsonUtility.FromJson<GameSaveData>(json);
                Debug.Log($"[FirebaseSync] Save pulled from cloud (version={data.saveVersion}).");
                return data;
            }
            catch (Exception e)
            {
                Debug.LogWarning($"[FirebaseSync] Pull failed: {e.Message}");
                return null;
            }
        }
    }
}
```

**3. Create `GoogleOAuthDesktop.cs`** at `StellarCommand/Assets/_Project/Scripts/Firebase/GoogleOAuthDesktop.cs`:

```csharp
namespace StellarCommand.Core
{
    using System;
    using System.Net;
    using System.Text;
    using System.Security.Cryptography;
    using System.Threading.Tasks;
    using UnityEngine;
    using UnityEngine.Networking;

    /// <summary>
    /// Custom OAuth 2.0 flow for Windows desktop.
    /// Opens browser -> user signs in -> local HTTP listener catches redirect -> exchanges code for ID token.
    /// Uses PKCE (no client secret embedded in binary).
    /// Per Research Pitfall 2: google-signin-unity is deprecated; this is the recommended approach.
    /// </summary>
    public static class GoogleOAuthDesktop
    {
        // These must be set before calling GetIdTokenAsync().
        // Load from a config file or ScriptableObject — do NOT hardcode secrets.
        public static string ClientId = "";
        private const string RedirectUri = "http://localhost:8856/callback";
        private const string AuthEndpoint = "https://accounts.google.com/o/oauth2/v2/auth";
        private const string TokenEndpoint = "https://oauth2.googleapis.com/token";
        private const int ListenerTimeoutSeconds = 120;

        public static async Task<string> GetIdTokenAsync()
        {
            if (string.IsNullOrEmpty(ClientId))
            {
                Debug.LogError("[GoogleOAuth] ClientId not configured.");
                return null;
            }

            string codeVerifier = GeneratePKCEVerifier();
            string codeChallenge = GeneratePKCEChallenge(codeVerifier);

            // 1. Open browser to Google consent screen
            string authUrl = $"{AuthEndpoint}?" +
                $"client_id={Uri.EscapeDataString(ClientId)}" +
                $"&redirect_uri={Uri.EscapeDataString(RedirectUri)}" +
                $"&response_type=code" +
                $"&scope=openid%20email%20profile" +
                $"&code_challenge={codeChallenge}" +
                $"&code_challenge_method=S256";

            System.Diagnostics.Process.Start(new System.Diagnostics.ProcessStartInfo
            {
                FileName = authUrl,
                UseShellExecute = true
            });

            // 2. Listen for redirect callback
            string code;
            using (var listener = new HttpListener())
            {
                listener.Prefixes.Add(RedirectUri + "/");
                listener.Start();

                var contextTask = listener.GetContextAsync();
                var timeoutTask = Task.Delay(TimeSpan.FromSeconds(ListenerTimeoutSeconds));
                var completed = await Task.WhenAny(contextTask, timeoutTask);

                if (completed == timeoutTask)
                {
                    Debug.LogError("[GoogleOAuth] Listener timed out waiting for callback.");
                    listener.Stop();
                    return null;
                }

                var context = await contextTask;
                code = context.Request.QueryString["code"];

                // Send response to browser
                byte[] buffer = Encoding.UTF8.GetBytes(
                    "<html><body><h2>Sign-in complete!</h2><p>You can close this tab and return to Stellar Command.</p></body></html>");
                context.Response.ContentLength64 = buffer.Length;
                context.Response.ContentType = "text/html";
                await context.Response.OutputStream.WriteAsync(buffer, 0, buffer.Length);
                context.Response.Close();
            }

            if (string.IsNullOrEmpty(code))
            {
                Debug.LogError("[GoogleOAuth] No authorization code received.");
                return null;
            }

            // 3. Exchange code for tokens (PKCE — no client secret needed)
            string tokenRequestBody = $"code={Uri.EscapeDataString(code)}" +
                $"&client_id={Uri.EscapeDataString(ClientId)}" +
                $"&redirect_uri={Uri.EscapeDataString(RedirectUri)}" +
                $"&grant_type=authorization_code" +
                $"&code_verifier={codeVerifier}";

            using var request = new UnityWebRequest(TokenEndpoint, "POST");
            byte[] bodyRaw = Encoding.UTF8.GetBytes(tokenRequestBody);
            request.uploadHandler = new UploadHandlerRaw(bodyRaw);
            request.downloadHandler = new DownloadHandlerBuffer();
            request.SetRequestHeader("Content-Type", "application/x-www-form-urlencoded");

            await request.SendWebRequest();

            if (request.result != UnityWebRequest.Result.Success)
            {
                Debug.LogError($"[GoogleOAuth] Token exchange failed: {request.error}");
                return null;
            }

            // Parse id_token from JSON response
            string responseJson = request.downloadHandler.text;
            var tokenResponse = JsonUtility.FromJson<TokenResponse>(responseJson);

            if (string.IsNullOrEmpty(tokenResponse.id_token))
            {
                Debug.LogError("[GoogleOAuth] No id_token in response.");
                return null;
            }

            return tokenResponse.id_token;
        }

        [Serializable]
        private class TokenResponse
        {
            public string id_token;
            public string access_token;
            public string token_type;
            public int expires_in;
        }

        private static string GeneratePKCEVerifier()
        {
            byte[] bytes = new byte[32];
            using var rng = RandomNumberGenerator.Create();
            rng.GetBytes(bytes);
            return Convert.ToBase64String(bytes)
                .TrimEnd('=')
                .Replace('+', '-')
                .Replace('/', '_');
        }

        private static string GeneratePKCEChallenge(string verifier)
        {
            using var sha256 = SHA256.Create();
            byte[] hash = sha256.ComputeHash(Encoding.ASCII.GetBytes(verifier));
            return Convert.ToBase64String(hash)
                .TrimEnd('=')
                .Replace('+', '-')
                .Replace('/', '_');
        }
    }
}
```

```csharp
namespace StellarCommand.Core
{
    using UnityEngine;
    using UnityEngine.UI;
    using TMPro;

    /// <summary>
    /// UI for the "Link Account" flow in Settings.
    /// Shows current auth status, Link/Unlink button, and status messages.
    /// Per D-02: optional Google Sign-In upgrade; per FOUND-12: Link Account in Settings.
    /// </summary>
    public class LinkAccountPanel : MonoBehaviour
    {
        [Header("UI Elements")]
        [SerializeField] private Button _linkButton;
        [SerializeField] private TextMeshProUGUI _statusText;
        [SerializeField] private TextMeshProUGUI _buttonText;
        [SerializeField] private GameObject _loadingIndicator;

        private bool _isLinking;

        private void OnEnable()
        {
            UpdateUI();
            _linkButton.onClick.AddListener(OnLinkButtonClicked);
        }

        private void OnDisable()
        {
            _linkButton.onClick.RemoveListener(OnLinkButtonClicked);
        }

        private void UpdateUI()
        {
            if (FirebaseAuthManager.Instance == null || !FirebaseAuthManager.Instance.IsAuthenticated)
            {
                _statusText.text = "Not connected to cloud services.";
                _buttonText.text = "Link Google Account";
                _linkButton.interactable = false;
                return;
            }

            if (FirebaseAuthManager.Instance.IsAnonymous)
            {
                _statusText.text = "Playing as guest. Link a Google account to back up your progress and restore on reinstall.";
                _buttonText.text = "Link Google Account";
                _linkButton.interactable = !_isLinking;
            }
            else
            {
                _statusText.text = $"Linked to Google account. Your progress is backed up to the cloud.";
                _buttonText.text = "Account Linked";
                _linkButton.interactable = false;
            }
        }

        private async void OnLinkButtonClicked()
        {
            _isLinking = true;
            _linkButton.interactable = false;
            _statusText.text = "Opening browser for Google Sign-In...";
            if (_loadingIndicator != null) _loadingIndicator.SetActive(true);

            try
            {
                string idToken = await GoogleOAuthDesktop.GetIdTokenAsync();

                if (string.IsNullOrEmpty(idToken))
                {
                    _statusText.text = "Sign-in was cancelled or timed out. Try again.";
                    return;
                }

                _statusText.text = "Linking account...";
                bool success = await FirebaseAuthManager.Instance.LinkWithGoogleAsync(idToken);

                if (success)
                {
                    _statusText.text = "Account linked successfully! Your progress is now backed up.";
                    // Trigger a save to push current data to cloud under the linked account
                    SaveManager.Instance?.Save();
                }
                else
                {
                    _statusText.text = "Linking failed. Your local progress is safe. Try again later.";
                }
            }
            catch (System.Exception e)
            {
                Debug.LogError($"[LinkAccount] Error: {e.Message}");
                _statusText.text = "An error occurred. Your local progress is safe.";
            }
            finally
            {
                _isLinking = false;
                if (_loadingIndicator != null) _loadingIndicator.SetActive(false);
                UpdateUI();
            }
        }
    }
}
```
